    int n,m,k;
    cin>>n>>m>>k;
    string s;
    cin>>s;
    int jump=0,swim=0;
    for(int i=0;i<n;i++){
        if(s[i]=='L'){
            jump=0;
        }
        else if(s[i]=='W'){
            if(jump!=m)jump++;
            if(jump==m)swim++;
        }else if(s[i]=='C'){
            jump++;
            if(jump==m){
                cout<<"NO"<<endl;
            return;}
        }
        
        if(jump>m || swim>k){
            cout<<"NO"<<endl;
            return;
        }
 
    }
    cout<<"YES"<<endl;
